    void bubbleSort(int arr[]) {
    boolean didSwap;
    for(int i = 0, len = arr.length; i < len - 1; i++) {
        didSwap = false;
        for(int j = 0; j < len - i - 1; j++) {
            if(arr[j + 1] < arr[j]) {
                swap(arr, j, j + 1);
                didSwap = true;
            }
        }
        if(didSwap == false)
            return;
    }    
}

void selectionSort(int* arr,int len){
    int min_index;
    for(int i=0;i<len;i++){
        min_index=i;
        for(int j=i;j<len;j++){
            if(arr[j]<arr[min_index]){

}

void merge(int* arr,int left,int right,int mid){
    int lIndex=left;
    int rIndex=mid+1;
    int help[right-left+1];
    int i=0;
    while(lIndex<=mid&&rIndex<=right){
        help[i++]=arr[lIndex]<arr[rIndex]?arr[lIndex++]:arr[rIndex++];
    }
    while(lIndex<=mid){
        help[i++]=arr[lIndex++];
    }
    while(rIndex<=right){
        help[i++]=arr[rIndex++];
    }
    for(int i=0;i<right-left+1;i++)
    {
        arr[left+i]=help[i];
    }
    
}

void CreateHeap(vector<int> &arr,int len,int index){
    int node=index;
    int left=2*index+1;//å·¦å­èŠ‚ç‚¹
    int right=2*index+2;//å³å­èŠ‚ç‚¹
    if(left<len&&arr[left]>arr[index]) index=left;
    if(right<len&&arr[right]>arr[index])index=right;
    if(index!=node){
        swap(arr[index],arr[node]);
        //å†å¾€ä¸‹éå†æ˜¯å¦æœ‰å­æ ‘å¯ä»¥é‡æ–°æ’åº
        CreateHeap(arr, len, index);
    }
}

        swap(arr[i],arr[0]);

        CreateHeap(arr, i, 0);
    }
}

void CountingSort(vector<int>& v,int maxId,int len){
    vector<int> temp;
    temp.reserve(maxId+1);//é¢„å…ˆä½¿æ•°ç»„ä¿ç•™maxidçš„ç©ºé—´
    temp.assign(maxId+1, 0);//ä½¿æ•°ç»„å€¼åˆå§‹åŒ–ä¸º0
    for(int i=0;i<len;i++){
            temp[v[i]]++;
    }
    //å°†tempä¸­çš„æ•°æ¬åˆ°åŸæ•°ç»„ä¸­
    int j=0;
    for(int i=0;i<maxId+1;i++){

            temp[i]--;
        }
    }
    
}
